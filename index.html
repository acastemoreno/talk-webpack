Entry

Entry

Punto de entrada en su funcionalidad como bundler. Solo archivos js o json 

# webpack.config.js
module.exports = {
  entry: './src/index.js' # Default setting
};
# webpack.config.js
module.exports = {
  entry: {
    app: './src/app.js',
    adminApp: './src/adminApp.js'
  }
};

Entry: todos los archivos de una carpeta

# webpack.config.js
const glob = require("glob");

module.exports = {
  entry: glob.sync("./src/js/*.js"),
};

Output

Output

Resultado final de la conversión de archivos indicados en entry 

# webpack.config.js
module.exports = {
  output: {
    filename: './dist/main.js' # Default setting
  } 
};
# webpack.config.js
const path = require('path');

module.exports = {
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'holi-woli.js'
  }
};

output: por cada archivo en entry

# webpack.config.js
const path = require("path");

module.exports = {
  output: {
    filename: "js/[name].js",
    path: path.resolve(__dirname, "dist"),
  },
};

fix: multiple output files

const glob = require("glob");
const path = require("path");

function toObject(paths) {
  var ret = {};
  paths.forEach(function (path) {
    // you can define entry names mapped to [name] here
    ret[path.split(/\/|\./).slice(-2)[0]] = path;
  });
  return ret;
}

module.exports = {
  entry: toObject(glob.sync("./src/js/*.js")),
  output: {
    filename: "js/[name].js",
    path: path.resolve(__dirname, "dist"),
  },
};

Loader

Loader

Webpack out-of-box solo comprende JavaScript (ES5) y JSON. Los loaders permiten a webpack entender y procesar otro tipo de archivos. 

# webpack.config.js
module.exports = {
  module: {
    rules: [
      { test: /\.txt$/, use: 'raw-loader' }
    ]
  }
};
# usage in any .js
import txt from 'raw-loader!./file.txt';

Process Images

$ yarn add file-loader --dev
module.exports = {
  # ...
  module: {
    rules: [
      {
        test: /\.(png|svg|jpg|gif)$/,
        use: [
          { loader: "file-loader", options: { outputPath: "images" }},
        ],
      },
    ],
  },
};

Plugin

Plugins

Plugins extiende las funcionalidades de Webpack, liberando la posibilidad de realizar diferentes tareas como optimización, asset management y inyeccion de variables de entorno.

Algunos plugins que resultan utiles:

  • HtmlWebpackPlugin: Simplifica el procesado de HTML files. Especialmente util cuando los archivos resultantes tiene algun hash en su nombre.
  • MiniCssExtractPlugin: Los css importados en .js son extraidos en archivos separados.
 

$ yarn add html-webpack-plugin html-loader --dev
const HtmlWebPackPlugin = require("html-webpack-plugin");

const htmls = glob.sync("./src/*.html").map((item) => {
  const name_file = path.basename(item, path.extname(item));

  return new HtmlWebPackPlugin({
    filename: `${name_file}.html`, template: item, chunks: [name_file] 
  });
});

module.exports = {
  module: {
    rules: [
      {
        test: /\.html$/,
        use: [
          { loader: "html-loader", options: { minimize: true } }
        ],
      },
    ],
  },
  plugins: htmls,
};
 

$ yarn add mini-css-extract-plugin css-loader sass-loader node-sass style-loader --dev
const MiniCssExtractPlugin = require("mini-css-extract-plugin");

module.exports = {
  module: {
    rules: [
      {
        test: /\.[s]?css$/,
        use: [MiniCssExtractPlugin.loader, "css-loader", "sass-loader"],
      },
    ],
  },
  plugins: [
    new MiniCssExtractPlugin({
      template: `./src/styles/[name].scss`,
      filename: "./styles/[name].css",
    }),
  ].concat(htmls),
};
